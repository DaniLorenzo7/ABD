CREATE TABLE PERSONAL
(id_personal NUMBER(5) CONSTRAINT personal_cp PRIMARY KEY,
nombre VARCHAR2(10),
apellidos VARCHAR2(30),
puesto VARCHAR2(9),
salario NUMBER(7,2));


CREATE TABLE HOTEL
(id_hotel NUMBER(4) CONSTRAINT hotel_cp PRIMARY KEY,
localizacion VARCHAR2(20),
aforoMax NUMBER(4),
eventos VARCHAR2(30),
FOREIGN KEY (director) REFERENCES PERSONAL (id_personal));


CREATE TABLE HORAS_TRABAJADAS
(id_personal NOT NULL NUMBER(5),
id_hotel NOT NULL NUMBER(4),
horas NUMBER(10),
PRIMARY KEY (id_personal, id_hotel));

CREATE TABLE RESERVA
(id_reserva NUMBER(10),
id_habitacion NUMBER(4),
